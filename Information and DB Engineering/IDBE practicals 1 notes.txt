CREATE TABLE Customer (
	customerName VARCHAR(50) PRIMARY KEY,
	street VARCHAR(50),
	customerCity(50)
);
CREATE TABLE Deposit ( 
	customerName VARCHAR(50),
	branchName VARCHAR(50),
	accountNumber INT PRIMARY KEY,
	balanace DECIMAL(10,2)
);
CREATE TABLE Loan (
	customerName VARCHAR(50),
	branchName VARCHAR(50),
	loanNumber INT PRIMARY KEY,
	amount DECIMAL(10,2)
);
CREATE TABLE Branch (
	branchName VARCHAR(50) PRIMARY KEY,
	branchCity VARCHAR(50),
	Assets INT
);
ALTER TABLE Deposit 
	ADD CONSTRAINT CustomerDeposit
	FOREIGN KEY (CustomerName)
	REFERENCES Customer(CustomerName);
ALTER TABLE Deposit
	ADD CONSTRAINT BranchDeposit
	FOREIGN KEY (branchName)
	REFERENCES Branch(branchName);
ALTER TABLE Loan
	ADD (
		CONSTRAINT BranchLoan
		FOREIGN KEY (branchName)
		REFERENCES Branch(branchName),
		CONSTRAINT CustomerLoan
		FOREIGN KEY (customerName)
		REFERENCES Customer(customerName)
	);
